```python
def remove_duplicates(a):
    return list(set(a))

string = 'aboba'
print(remove_duplicates(string)) # ['a', 'b', 'o']

arr = [1,2,3,4,4,4,5,6,1,3,6]
print(remove_duplicates(arr)) # [1, 2, 3, 4, 5, 6]
```

- –ü–æ–ª—É—á–µ–Ω–Ω–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –ø–µ—Ä–µ–≤–æ–¥–∏—Ç –≤ —Å–ø–∏—Å–æ–∫ (–¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –¥–∞–ª—å–Ω–µ–π—à–µ–π —Ä–∞–±–æ—Ç—ã –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è)

**–ú–æ–∂–Ω–æ —Å–¥–µ–ª–∞—Ç—å —Ñ—É–Ω–∫—Ü–∏—é –ø–æ–∏–Ω—Ç–µ—Ä–µ—Å–Ω–µ–π:**

```python
def remove_duplicates(a):
    result = list(dict.fromkeys(a))  # —Å–æ—Ö—Ä–∞–Ω—è–µ—Ç –ø–æ—Ä—è–¥–æ–∫ –∏ —É–±–∏—Ä–∞–µ—Ç –¥—É–±–ª–∏–∫–∞—Ç—ã

    if isinstance(a, str):   # –µ—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ ‚Üí –≤–µ—Ä–Ω—É—Ç—å —Å—Ç—Ä–æ–∫—É
        return ''.join(result)
    elif isinstance(a, tuple):  # –µ—Å–ª–∏ –∫–æ—Ä—Ç–µ–∂ ‚Üí –≤–µ—Ä–Ω—É—Ç—å –∫–æ—Ä—Ç–µ–∂
        return tuple(result)
    else:  # –∏–Ω–∞—á–µ –æ—Å—Ç–∞–≤–∏—Ç—å —Å–ø–∏—Å–æ–∫
        return result


string = 'aboba'
print(remove_duplicates(string))  # 'abob'

arr = [1,2,3,4,4,4,5,6,1,3,6]
print(remove_duplicates(arr))  # [1, 2, 3, 4, 5, 6]

tpl = (1,2,2,3,1,4)
print(remove_duplicates(tpl))  # (1, 2, 3, 4)

```

–§—É–Ω–∫—Ü–∏—è `remove-duplicates`:

- –ü–µ—Ä–µ–≤–æ–¥–∏—Ç –ø–æ–ª—É—á–µ–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –≤ —Å–ª–æ–≤–∞—Ä—å (—Ç–∞–∫ –∫–∞–∫ –æ–Ω –±–æ–ª–µ–µ –æ—Ä–≥–∞–Ω–∏–∑–æ–≤–∞–Ω, —á–µ–º set)
- –ü–æ–ª—É—á–µ–Ω–Ω—ã–π —Å–ª–æ–≤–∞—Ä—å –ø–µ—Ä–µ–≤–æ–¥–∏—Ç –≤ list –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞
